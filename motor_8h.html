Detailed Description

A simple motor control class.

Example usage

#include "motor.h"

MOTOR m;

int main () {
    DDRD |= _BV(PORTD4);
    motor_new(&m, &PORTD, &OCR0A, PORTD4, false);
    while (1) {
        motor_forward(&m, 127);
        _delay_ms(1000);
        motor_stop(&m);
        _delay_ms(1000);
        motor_reverse(&m, 127);
        _delay_ms(1000);
        motor_stop(&m);
        _delay_ms(1000);
}
